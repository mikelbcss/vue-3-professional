```ts
// useStepCounter.ts
import { computed, ref } from 'vue'

export interface UseStepCounterOptions {
  initial: number
  step?: number
}

export function useStepCounter(options: UseStepCounterOptions) {
  const step = options.step ?? 1
  const count = ref<number>(options.initial)

  function set(n: number) {
    count.value = n
  }

  function increment() {
    set(count.value + step)
  }

  function decrement() {
    set(count.value - step)
  }

  function reset() {
    set(options.initial)
  }

  const isEven = computed(() => count.value % 2 === 0)

  return { count, increment, decrement, reset, isEven }
}
```

      <span class="text-xs text-muted-foreground" v-if="isEven">Es par</span>
    </div>
    <div class="flex gap-2">
      <button class="px-3 py-1 rounded bg-primary text-primary-foreground" @click="increment">+{{ step }}</button>
      <button class="px-3 py-1 rounded bg-secondary text-secondary-foreground" @click="decrement">-{{ step }}</button>
      <button class="px-3 py-1 rounded border" @click="reset">Reset</button>
    </div>
  </div>
</template>

<script setup lang="ts">
import { useStepCounter } from './useStepCounter'

  step,
})
</script>

> Nota: Cada uso de `useStepCounter()` crea un estado independiente. No hay límites `min/max` ni modos de normalización; solo `initial`, `step`, `increment`, `decrement` y `reset`.
